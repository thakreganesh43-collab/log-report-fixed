import json, os
REPORT_PATH = "/app/report.json"

def load_report():
    with open(REPORT_PATH) as f:
        return json.load(f)

def test_report_exists():
    assert os.path.exists(REPORT_PATH)

def test_report_valid_json():
    report = load_report()
    assert isinstance(report, dict)

def test_report_keys_present():
    report = load_report()
    for key in ["summary", "entries"]:
        assert key in report

def test_entries_non_empty_list():
    report = load_report()
    assert isinstance(report["entries"], list)
    assert len(report["entries"]) > 0

def test_summary_total_matches_entries():
    report = load_report()
    assert "total" in report["summary"]
    assert report["summary"]["total"] == len(report["entries"])
